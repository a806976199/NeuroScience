\section{2.3}
\label{sec:2.3}

\begin{prop}
  The mass of fluid in a region $W$ at time $t$ is
  \begin{equation}
    \label{eq:totalMass}
    m(W,t) = \int_W\rho(\mathbf{x},t)\dif V,
  \end{equation}
  where $\dif V$ is the area element in the plane
  or the volume element in space.
\end{prop}
